// SCORE REPORTING //

FUNCTION calculate_scores(user_answers, question_data):
    INITIALIZE category_totals = {}
    INITIALIZE identified_gaps = []

    FOR answer IN user_answers:
        weight = question_data[answer.id].weight
        score = answer.value
        category = question_data[answer.id].category

        // Weight calculation
        category_totals{category} += (score * weight)

        // Gap Detection
        IF score <= 2:
            identified_gaps.append(question_data[answer.id].gap_description)

    RETURN category_totals, identified_gaps

FUNCTION generate_action_plan(gaps):
    // Get recommendations based on gap IDs
    action_plan = DB.query("SELECT rec FROM recommendations WHERE gap_id IN gaps")
    RETURN action_plan
